brief Crea el filtro de dilatacion

void Filtros::Dilatacion(){ Mat imagen = matrizImagen; Mat nueva(imagen.rows, imagen.cols, CV_8UC3); Vec3b pixel; Vec3b blanco = {255,255,255}; Vec3b negro = {0,0,0}; for (int i=0; i<imagen.rows; i++){ for (int j=0; j<imagen.cols; j++){ pixel=imagen.at<Vec3b>(i, j); double gris = pixel[0]*0.3 + pixel[1]*0.59+pixel[2]*0.11; if (gris >127){ nueva.at<Vec3b>(i, j)=blanco; }else{ nueva.at<Vec3b>(i, j)=negro; } } }
